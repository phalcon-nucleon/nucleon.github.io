Array

Arr::accessible

Determine whether the given value is array accessible.

Arr::accessible($value)

Arr::add

Add an element to an array using "dot" notation if it doesn't exist.

Arr::add($array, $key, $value)

Arr::collapse

Collapse an array of arrays into a single array.

Arr::collapse($array)

Arr::divide

Divide an array into two arrays. One with keys and the other with values.

Arr::divide($array)

Arr::dot

Flatten a multi-dimensional associative array with dots.

Arr::dot($array, $prepend = '')

Arr::except

Get all of the given array except for a specified array of items.

Arr::except($array, $keys)

Arr::exists

Determine if the given key exists in the provided array.

Arr::exists($array, $key)

Arr::first

Return the first element in an array passing a given truth test.

Arr::first($array, $callback = null, $default = null)

Arr::last

Return the last element in an array passing a given truth test.

Arr::last($array, callable $callback = null, $default = null)

Arr::flatten

Flatten a multi-dimensional array into a single level.

Arr::flatten($array, $depth = INF)

Arr::forget

Remove one or many array items from a given array using "dot" notation.

Arr::forget(&$array, $keys)

Arr::read

Get an item from an array.

Arr::read($array, $key, $default = null)

Arr::fetch

Get an item, no null, from an array.

Arr::fetch($array, $key, $default = null)

Arr::get

Get an item from an array using "dot" notation.

Arr::get($array, $key, $default = null)

Arr::has

Check if an item exists in an array using "dot" notation.

Arr::has($array, $keys)

Arr::isAssoc

Determines if an array is associative.
An array is "associative" if it doesn't have sequential numerical keys beginning with zero.

Arr::isAssoc(array $array)

Arr::only

Get a subset of the items from the given array.

Arr::only($array, $keys)

Arr::pluck

Pluck an array of values from an array.

Arr::pluck($array, $value, $key = null)

Arr::prepend

Push an item onto the beginning of an array.

Arr::prepend($array, $value, $key = null)

Arr::pull

Get a value from the array, and remove it.

Arr::pull(&$array, $key, $default = null)

Arr::set

Set an array item to a given value using "dot" notation.
If no key is given to the method, the entire array will be replaced.

Arr::set(&$array, $key, $value)

Arr::sortRecursive

Recursively sort an array by keys and values.

Arr::sortRecursive($array)

Arr::where

Filter the array using the given callback.

Arr::where($array, callable $callback)

Arr::map

Map an array recursively

Arr::map($callback, array $array, $recursive = false)

Object

Obj::fill

Fill in data where it's missing.

Obj::fill(&$target, $key, $value)

Obj::read

Obj::read($object, $property, $default = null)

Obj::fetch

Obj::fetch($object, $property, $default = null)

Obj::get

Get an item from an array or object using "dot" notation.

Obj::get($target, $key, $default = null)

Obj::set

Set an item on an array or object using dot notation.

Obj::set(&$target, $key, $value, $overwrite = true)

Obj::value

Return the default value of the given value.

Obj::value($value)

String

Str::ascii

Transliterate a UTF-8 value to ASCII.

Str::ascii($value)

Str::camel

Convert a value to camel case.

Str::camel($value)

Str::contains

Determine if a given string contains a given substring.

Str::contains($haystack, $needles)

Str::endsWith

Determine if a given string ends with a given substring.

Str::endsWith($haystack, $needles)

Str::finish

Cap a string with a single instance of a given value.

Str::finish($value, $cap)

Str::is

Determine if a given string matches a given pattern.

Str::is($pattern, $value)

Str::length

Return the length of the given string.

Str::length($value)

Str::levenshtein

Str::levenshtein($word, $words, $order = SORT_ASC, $sort_flags = SORT_REGULAR)

Str::limit

Limit the number of characters in a string.

Str::limit($value, $limit = 100, $end = '...')

Str::lower

Convert the given string to lower-case.

Str::lower($value)

Str::words

Limit the number of words in a string.

Str::words($value, $words = 100, $end = '...')

Str::parseCallback

Parse a Class::method style callback into class and method.

Str::parseCallback($callback, $default)

Str::random

Generate a more truly "random" alpha-numeric string.

Str::random($length = 16)

Str::quickRandom

Generate a "random" alpha-numeric string. Should not be considered sufficient for cryptography, etc.

Str::quickRandom($length = 16)

Str::replaceFirst

Replace the first occurrence of a given value in the string.

Str::replaceFirst($search, $replace, $subject)

Str::replaceLast

Replace the last occurrence of a given value in the string.

Str::replaceLast($search, $replace, $subject)

Str::upper

Convert the given string to upper-case.

Str::upper($value)

Str::title

Convert the given string to title case.

Str::title($value)

Str::slug

Generate a URL friendly "slug" from a given string.

Str::slug($title, $separator = '-')

Str::snake

Convert a string to snake case.

Str::snake($value, $delimiter = '_')

Str::startsWith

Determine if a given string starts with a given substring.

Str::startsWith($haystack, $needles)

Str::studly

Convert a value to studly caps case.

Str::studly($value)

Str::capitalize

Convert a value to Capitalize case.

Str::capitalize($value)

Str::substr

Returns the portion of string specified by the start and length parameters.

Str::substr($string, $start, $length = null)

Str::ucfirst

Make a string's first character uppercase.

Str::ucfirst($string)
