Array

Arr::accessible

Determine whether the given value is array accessible.

$value = [];
Arr::accessible([]);
// true

$value = new class implements ArrayAccess{};
Arr:accessible($value);
// true

Arr::add

Add an element to an array using "dot" notation if it doesn't exist.

$array = Arr::add([], 'foo.bar', 123);

// ['foo' => ['bar' => 123]]

Arr::collapse

Collapse an array of arrays into a single array.

$array = Arr::collapse([[1, 2, 3], [4, 5, 6], [7, 8, 9]]);

// [1, 2, 3, 4, 5, 6, 7, 8, 9]

Arr::divide

Divide an array into two arrays. One with keys and the other with values.

[$keys, $values] = Arr::divide(['name' => 'Desk']);

// $keys: ['name']

// $values: ['Desk']

Arr::dot

Flatten a multi-dimensional associative array with dots.

$array = ['products' => ['desk' => ['price' => 100]]];

$flattened = Arr::dot();

// ['products.desk.price' => 100]

Arr::except

Get all of the given array except for a specified array of items.

$array = ['name' => 'Desk', 'price' => 100];

$filtered = Arr::except($array, ['price']);

// ['name' => 'Desk']

Arr::exists

Determine if the given key exists in the provided array.

$array = ['name' => 'Desk', 'price' => 100];

Arr::exists($array, 'price');

// true

Arr::first

Return the first element in an array passing a given truth test.

$array = [100, 200, 300];

$first = Arr::first($array, function ($value, $key) {
    return $value >= 150;
});

// 200

Arr::last

Return the last element in an array passing a given truth test.

$array = [100, 200, 300, 110];

$first = Arr::last($array, function ($value, $key) {
    return $value >= 150;
});

// 300

Arr::flatten

Flatten a multi-dimensional array into a single level.

$array = ['name' => 'Joe', 'languages' => ['PHP', 'Ruby']];

$flattened = Arr::flatten($array);

// ['Joe', 'PHP', 'Ruby']

Arr::forget

Remove one or many array items from a given array using "dot" notation.

$array = ['products' => ['desk' => ['price' => 100]]];

Arr::forget($array, 'products.desk');

// ['products' => []]

Arr::read

Get an item from an array.

$array = ['foo' => 123, 'bar' => null];

$value = Arr::read($array, 'foo', 'default');

// 123

$value = Arr::read($array, 'bar', 'default');

// null

$value = Arr::read($array, 'baz', 'default');

// 'default'

Arr::fetch

Get an item, no null, from an array.

$array = ['foo' => 123, 'bar' => null];

$value = Arr::read($array, 'foo', 'default');

// 123

$value = Arr::read($array, 'bar', 'default');

// 'default'

$value = Arr::read($array, 'baz', 'default');

// 'default'

Arr::get

Get an item from an array using "dot" notation.

$array = ['products' => ['desk' => ['price' => 100]]];

$price = Arr::get($array, 'products.desk.price');

// 100

Arr::has

Check if an item exists in an array using "dot" notation.

$array = ['product' => ['name' => 'desk', 'price' => 100]];

$hasItem = Arr::has($array, 'product.name');

// true

$hasItems = Arr::has($array, ['product.price', 'product.discount']);

// false

Arr::isAssoc

Determines if an array is associative.
An array is "associative" if it doesn't have sequential numerical keys beginning with zero.

$value = Arr::isAssoc(['a', 'b', 'c']);

// true

$value = Arr::isAssoc(['a' => 1, 'b' => 2, 'c' => 3);

// false

Arr::only

Get a subset of the items from the given array.

$array = ['name' => 'Desk', 'price' => 100, 'orders' => 10];

$array = Arr::only($array, ['name', 'price']);

// ['name' => 'Desk', 'price' => 100]

Arr::pluck

Pluck an array of values from an array.

$array = [
    ['developer' => ['id' => 1, 'name' => 'Ark']],
    ['developer' => ['id' => 2, 'name' => 'Maryline']],
];

$array = Arr::pluck($array, 'developer.name');

// ['Ark', 'Maryline'];
You may also specify how you wish the resulting list to be keyed: 
$array = Arr::pluck($array, 'developer.name', 'developer.id');

// [1 => 'Ark', 2 => 'Maryline'];

Arr::prepend

Push an item onto the beginning of an array.

$array = ['one', 'two', 'three', 'four'];

$array = Arr::prepend($array, 'zero');

// $array: ['zero', 'one', 'two', 'three', 'four']

Arr::pull

Get a value from the array, and remove it.

$array = ['name' => 'Desk', 'price' => 100];

$name = Arr::pull($array, 'name');

// $name: Desk

// $array: ['price' => 100]

Arr::set

Set an array item to a given value using "dot" notation.
If no key is given to the method, the entire array will be replaced.

$array = ['products' => ['desk' => ['price' => 100]]];

Arr::set($array, 'products.desk.price', 200);

// ['products' => ['desk' => ['price' => 200]]]

Arr::sortRecursive

Recursively sort an array by keys and values.

$array = [
    [
        'Roman',
        'Ark',
        'Li',
    ],
    [
        'PHP',
        'Ruby',
        'JavaScript',
    ],
];

$array = Arr::sortRecursive($array);

/*
    [
        [
            'Li',
            'Roman',
            'Ark',
        ],
        [
            'JavaScript',
            'PHP',
            'Ruby',
        ]
    ];
*/

Arr::where

Filter the array using the given callback.

$array = [100, '200', 300, '400', 500];

$array = Arr::where($array, function ($value, $key) {
    return is_string($value);
});

// [1 => 200, 3 => 400]

Arr::map

Map an array recursively

$array = ['foo' => ['Abc', 'dEf', 'GHI']];

$mapped = Arr::map('strtolower', $array);

// ['foo' => ['abc', 'def', 'ghi']]

Object

Obj::fill

Fill in data where it's missing.

$object = new stdClass();

Obj::fill($object, 'foo', 'bar');

// $object->foo = 'bar';

Obj::fill($object, 'foo', 'baz');

// $object->foo = 'bar';

Obj::read

Get an item from an object.

$object = new stdClass();
$object->foo = 'bar';
$object->baz = null;

$value = Obj::read($object, 'foo', 'default');

// 'bar'

$value = Obj::read($object, 'baz', 'default');

// null

$value = Obj::read($object, 'dot', 'default');

// 'default'

Obj::fetch

Get an item, no null, from an object.

$object = new stdClass();
$object->foo = 'bar';
$object->baz = null;

$value = Obj::fetch($object, 'foo', 'default');

// 'bar'

$value = Obj::fetch($object, 'baz', 'default');

// 'default'

$value = Obj::fetch($object, 'dot', 'default');

// 'default'

Obj::get

Get an item from an array or object using "dot" notation.

$object = new stdClass();
$object->foo = new stdClass();
$object->foo->bar = 'baz';

$value = Obj::get($object, 'foo.bar');

// 'baz'

Obj::set

Set an item on an array or object using dot notation.

$object = new stdClass();

Obj::set($object, 'foo.bar', 'baz');

// $object->foo->bar === 'baz'

Obj::value

Return the default value of the given value.

$value = Obj::value(123);

// 123

$value = Obj::value(function(){
    return 'abc';
});

// 'abc'

String

Str::ascii

Transliterate a UTF-8 value to ASCII.

$value = Str::ascii('se√±or');

// senor

Str::camel

Convert a value to camel case.

$value = Str::camel('foo_bar');

// fooBar

Str::contains

Determine if a given string contains a given substring.

$value = Str::contains('foo bar', 'foo');

// true

Str::endsWith

Determine if a given string ends with a given substring.

$result = Str::endsWith('foo bar', 'bar');

// true

Str::finish

Cap a string with a single instance of a given value.

$adjusted = Str::finish('this/string', '/');

// this/string/

$adjusted = Str::finish('this/string/', '/');

// this/string/

Str::is

Determine if a given string matches a given pattern.

$matches = Str::is('foo*', 'foobar');

// true

$matches = Str::is('baz*', 'foobar');

// false

Str::length

Return the length of the given string.

$length = Str::length('foo');

// 3

Str::levenshtein

Sort a words list by their levenshtein distance of word.

$words = Str::levenshtein('abc', ['aac', 'bca', 'abc', 'abd', 'xzy']);

/*
[
  'abc' => 0,
  'aac' => 1,
  'abd' => 1,
  'bca' => 2,
  'xzy' => 3,
]
*/

Str::limit

Limit the number of characters in a string.

$truncated = Str::limit('The quick brown fox jumps over the lazy dog', 20);

// The quick brown fox...

You may also pass a third argument to change the string that will be appended to the end:

$truncated = Str::limit('The quick brown fox jumps over the lazy dog', 20, ' (...)');

// The quick brown fox (...)

Str::lower

Convert the given string to lower-case.

$lower = Str::lower('Foo Bar');

// 'foo bar'

Str::words

Limit the number of words in a string.

$value = Str::words('foo bar baz', 2);

// 'foo bar...'

You may also pass a third argument to change the string that will be appended to the end:

$value = Str::words('foo bar baz', 2, '(...)');

// 'foo bar(...)'

Str::parseCallback

Parse a "Class@method" style callback into class and method.

$callable = Str::parseCallback("Foo@bar");

// ['Foo', 'bar']

Str::random

Generate a more truly "random" alpha-numeric string.
This function call random_bytes, or Sodium\randombytes_buf, or openssl_random_pseudo_bytes, or mcrypt_create_iv, or Phalcon\Security\Random::bytes 

$random = Str::random(16);

Str::quickRandom

Generate a "random" alpha-numeric string. Should not be considered sufficient for cryptography, etc.

$random = Str::quickRandom(16)

Str::replaceFirst

Replace the first occurrence of a given value in the string.

$replaced = Str::replaceFirst('the', 'a', 'the quick brown fox jumps over the lazy dog');

// a quick brown fox jumps over the lazy dog

Str::replaceLast

Replace the last occurrence of a given value in the string.

$replaced = Str::replaceLast('the', 'a', 'the quick brown fox jumps over the lazy dog');

// the quick brown fox jumps over a lazy dog

Str::upper

Convert the given string to upper-case.

$lower = Str::upper('Foo Bar');

// 'FOO BAR'

Str::slug

Generate a URL friendly "slug" from a given string.

$slug = Str::slug('Nucleon Framework', '-');

// nucleon-framework

Str::snake

Convert a string to snake case.

$lower = Str::snake('fooBar');

// 'foo_bar'

Str::studly

Convert a value to studly caps case.

$converted = Str::studly('foo_bar');

// FooBar

Str::capitalize

Convert a value to Capitalize case.

$converted = Str::capitalize('foo bar');

// 'Foo Bar'

Str::startsWith

Determine if a given string starts with a given substring.

$value = Str::startsWith('this/string', '/');

// false

$value = Str::startsWith('this/string', ['/', 't']);

// true

Str::title

Convert the given string to title case. This function use mb_convert_case.

$lower = Str::title('foo bar');

// 'Foo Bar'

Str::ucfirst

Make a string's first character uppercase.

Str::ucfirst($string)

Str::substr

Returns the portion of string specified by the start and length parameters.

Str::substr($string, $start, $length = null)
